- The Monotonic Stack approach is the best: O(n) time and O(n) space, and is the standard optimal solution for this problem.

# Brute Force Solution
n = int(input())
temps = list(map(int, input().split()))
res = [0] * n
for i in range(n):
    for j in range(i + 1, n):
        if temps[j] > temps[i]:
            res[i] = j - i
            break
print(' '.join(map(str, res)))

---

# Monotonic Stack Solution
n = int(input())
temps = list(map(int, input().split()))
res = [0] * n
stack = []
for i, temp in enumerate(temps):
    while stack and temp > temps[stack[-1]]:
        idx = stack.pop()
        res[idx] = i - idx
    stack.append(i)
print(' '.join(map(str, res)))

# Reverse Traversal with Preprocessing Solution
n = int(input())
temps = list(map(int, input().split()))
res = [0] * n
next_occurrence = [float('inf')] * 71  # temperatures from 30 to 100

for i in range(n - 1, -1, -1):
    min_wait = float('inf')
    for t in range(temps[i] + 1, 101):
        idx = next_occurrence[t - 30]
        if idx < min_wait:
            min_wait = idx
    if min_wait < float('inf'):
        res[i] = min_wait - i
    next_occurrence[temps[i] - 30] = i

print(' '.join(map(str, res)))
