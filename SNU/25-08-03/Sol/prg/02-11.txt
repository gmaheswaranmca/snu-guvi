--- Sorting Approach ---
def kth_largest_sorting(arr, k):
    arr.sort(reverse=True)
    return arr[k-1]

# Example usage:
n, k = map(int, input().split())
arr = list(map(int, input().split()))
print(kth_largest_sorting(arr, k))

--- Sorting Approach (Java) ---
import java.util.*;

public class KthLargestSorting {
    public static int kthLargestSorting(int[] arr, int k) {
        Arrays.sort(arr);
        return arr[arr.length - k];
    }

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        int k = sc.nextInt();
        int[] arr = new int[n];
        for(int i = 0; i < n; i++) {
            arr[i] = sc.nextInt();
        }
        System.out.println(kthLargestSorting(arr, k));
    }
}
